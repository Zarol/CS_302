00011 // ABC that gives an interface for the various search implementations
00012 // This allows each class to act like a function by defining
00013 // operator() which takes a search value and sorted list of keys to
00014 // look through.  We do this so that we can use STL functions (e.g.,
00015 // for_each) to // iterate through a list of keys to try since they
00016 // expect to call a function for each element.
00017 class Search : public binary_function<int, vector<int>, bool> {

00034 class binarySearch : public Search {
00035   public:
00036     bool operator()(int searchValue, const vector<int>& keys) const {
00037     int low = 0;
00038     int high = keys.size() - 1;
00039 
00040     while (low <= high) {
00041         int index = (low + high) / 2;
00042         int key = keys[index];
00043         if (searchValue < key) {
00044         high = index - 1;
00045         } else if (searchValue > key) {
00046         low = index + 1;
00047         } else {
00048         return true;
00049         }
00050     }

00052     return false;
00053     }
